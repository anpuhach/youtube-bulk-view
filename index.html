Run the code below on the playlist page to get the URLs

var list = document.getElementsByClassName("yt-simple-endpoint inline-block style-scope ytd-thumbnail");
var res = "";
for (let i of list) {
    if (i.href == "") continue;
    res += i.href.substring(0, i.href.indexOf("&")) + "\n";
}
console.log(res);
